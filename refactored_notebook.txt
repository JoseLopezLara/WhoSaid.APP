        "def extract_video_id(url):\n",
        "    \"\"\"Extrae el ID del video de YouTube desde la URL.\"\"\"\n",
        "    match = re.search(r'(?:v=|\\/)([0-9A-Za-z_-]{11}).*', url)\n",

        "    video_id = extract_video_id(url)\n",

        "        return\n",

        "    try:\n",
        "        # MODIFICATION: Use of the instance and fetch() method according to documentation\n",
        "        # languages=['es', 'en'] tries Spanish first, then English\n",
        "        fetched_transcript = ytt_api.fetch(video_id, languages=['es', 'en'])\n",
        "\n",
        "        # Convertimos el objeto FetchedTranscript a una lista de diccionarios (raw data)\n",
        "        transcript = fetched_transcript.to_raw_data()\n",

        "        creator_folder = os.path.join(\"processed_files\", \"processed_transcriptions\", creator)\n",

        "def clean_word(word):\n",
        "    \"\"\"Converts to lowercase and removes punctuation signs.\"\"\"\n",
        "    word = word.lower()\n",
        "    return word.translate(str.maketrans('', '', string.punctuation))\n",
        "\n",
        "def is_valid_ngram(ngram_words, use_stopwords_filter=True):\n",
        "    \"\"\"\n",
        "    Business rule: If the filter is active, the n-gram is discarded\n",
        "    SOLO si TODAS las words que lo componen son stopwords.\n",
        "    \"\"\"\n",
        "    if not use_stopwords_filter:\n",
        "        return True\n",
        "\n",
        "    # Validate if all words are in the stopwords list\n",
        "    are_all_stopwords = all(word in stop_words for word in ngram_words)\n",
        "\n",
        "    # It is valid if NOT all are stopwords (i.e., it has at least one valuable word)\n",
        "    return not are_all_stopwords\n",
        "\n",
        "def tokenize_transcript(transcript):\n",
        "    \"\"\"\n",
        "    Decomposes the transcript into a list of words where each word\n",
        "    retiene el inicio (start) y el fin (start + duration) de su block original.\n",
        "    \"\"\"\n",
        "    tokens = []\n",
        "    for block in transcript:\n",
        "        text = block.get('text', '')\n",
        "        start = block.get('start', 0)\n",
        "        end = start + block.get('duration', 0)\n",
        "\n",
        "        words = text.split()\n",
        "        for p in words:\n",
        "            clean_p = clean_word(p)\n",
        "            if clean_p: # Avoid empty strings\n",
        "                tokens.append({\n",
        "                    'word': clean_p,\n",
        "                    'start': start,\n",
        "                    'end': end\n",
        "                })\n",
        "    return tokens\n",
        "\n",
        "def extract_creator_ngrams(creator, n_phrases, use_stopwords_filter=True):\n",
        "    \"\"\"Processes all JSON files for a creator, generates n-grams and separates union/intersection.\"\"\"\n",
        "    creator_folder = os.path.join(\"processed_files\", \"processed_transcriptions\", creator)\n",
        "    if not os.path.exists(creator_folder):\n",
        "        print(f\"Folder not found for creator {creator}\")\n",
        "        return None, None\n",
        "\n",
        "    json_files = [f for f in os.listdir(creator_folder) if f.endswith('.json')]\n",
        "    total_videos = len(json_files)\n",
        "    if total_videos == 0:\n",
        "        return None, None\n",
        "\n",
        "    # { n_size: { \"ngrama_str\": {\"count\": int, \"video_appearances\": set(video_id), \"start\": float, \"end\": float, \"url\": str, \"appearances\": []} } }\n",
        "    creator_results = {i: {} for i in range(2, 7)}\n",
        "\n",
        "    for file_name in json_files:\n",
        "        file_path = os.path.join(creator_folder, file_name)\n",
        "        try:\n",
        "            with open(file_path, 'r', encoding='utf-8') as f:\n",
        "                data = json.load(f)\n",
        "\n",
        "            transcript = data['transcript']\n",
        "            url = data['url']\n",
        "            video_id = data['video_id']\n",
        "\n",
        "            tokens = tokenize_transcript(transcript)\n",
        "            total_tokens = len(tokens)\n",
        "\n",
        "            for n in range(2, 7):\n",
        "                for i in range(total_tokens - n + 1):\n",
        "                    ngram_tokens = tokens[i:i+n]\n",
        "                    ngram_words = tuple(t['word'] for t in ngram_tokens)\n",
        "\n",
        "                    if not is_valid_ngram(ngram_words, use_stopwords_filter):\n",
        "                        continue\n",
        "\n",
        "                    ngram_str = \" \".join(ngram_words)\n",
        "                    start_time = ngram_tokens[0]['start']\n",
        "                    end_time = ngram_tokens[-1]['end']\n",
        "                    appearance = {'start': start_time, 'end': end_time, 'url': url}\n",
        "\n",
        "                    if ngram_str in creator_results[n]:\n",
        "                        creator_results[n][ngram_str]['count'] += 1\n",
        "                        creator_results[n][ngram_str]['video_appearances'].add(video_id)\n",
        "                        creator_results[n][ngram_str]['appearances'].append(appearance)\n",
        "                    else:\n",
        "                        creator_results[n][ngram_str] = {\n",
        "                            'count': 1,\n",
        "                            'video_appearances': {video_id},\n",
        "                            'start': start_time,\n",
        "                            'end': end_time,\n",
        "                            'url': url,\n",
        "                            'appearances': [appearance]\n",
        "                        }\n",
        "        except Exception as e:\n",
        "            print(f\"Error processing file {file_path}: {str(e)}\")\n",
        "\n",
        "    # Separate into union and intersection\n",
        "    min_videos_required = total_videos * INTERSECTION_THRESHOLD\n",
        "    \n",
        "    union_results = {i: [] for i in range(2, 7)}\n",
        "    intersection_results = {i: [] for i in range(2, 7)}\n",
        "\n",
        "    for n in range(2, 7):\n",
        "        all_ngrams = []\n",
        "        intersect_ngrams = []\n",
        "        \n",
        "        for ngram_str, info in creator_results[n].items():\n",
        "            # Remove set before returning/sorting (sets are not JSON serializable)\n",
        "            videos_count = len(info['video_appearances'])\n",
        "            clean_info = {k: v for k, v in info.items() if k != 'video_appearances'}\n",
        "            \n",
        "            all_ngrams.append((ngram_str, clean_info))\n",
        "            if videos_count >= min_videos_required:\n",
        "                intersect_ngrams.append((ngram_str, clean_info))\n",
        "\n",
        "        # Sort and take top N\n",
        "        all_ngrams.sort(key=lambda x: x[1]['count'], reverse=True)\n",
        "        intersect_ngrams.sort(key=lambda x: x[1]['count'], reverse=True)\n",
        "        \n",
        "        union_results[n] = all_ngrams[:n_phrases]\n",
        "        intersection_results[n] = intersect_ngrams[:n_phrases]\n",
        "\n",
        "    return creator, {'union': union_results, 'intersection': intersection_results}\n",
        "\n",
        "def execute_stage_2(n_phrases=10, num_threads=4, use_stopwords_filter=True):\n",
        "    \"\"\"Empties the queue, processes in parallel by creator, and extracts the Top N phrases.\"\"\"\n",
        "    creators_to_process = []\n",
        "\n",
        "    # Empty the queue\n",
        "    while not processing_queue.empty():\n",
        "        creators_to_process.append(processing_queue.get())\n",
        "\n",
        "    if not creators_to_process:\n",
        "        print(\"The queue is empty. Execute Stage 1 first.\")\n",
        "        return None\n",
        "\n",
        "    print(f\"Starting Stage 2: Processing {len(creators_to_process)} creators with {num_threads} threads...\")\n",
        "\n",
        "    # Structure: { creator: {'union': {n: [(phrase, info)]}, 'intersection': {n: [(phrase, info)]}} }\n",
        "    top_results = {}\n",
        "\n",
        "    with concurrent.futures.ThreadPoolExecutor(max_workers=num_threads) as executor:\n",
        "        # Send tasks\n",
        "        futures = [executor.submit(extract_creator_ngrams, creator, n_phrases, use_stopwords_filter) for creator in creators_to_process]\n",
        "\n",
        "        # Collect results as they finish\n",
        "        for futuro in concurrent.futures.as_completed(futures):\n",
        "            creator, creator_results = futuro.result()\n",
        "            if creator and creator_results:\n",
        "                top_results[creator] = creator_results\n",
        "\n",
        "    print(\"Stage 2 finished.\")\n",
        "    return top_results\n",
        "\n",
        "def format_results(creator_results, mode):\n",
        "    \"\"\"Formats the ngrams results into either 'first_appearance' or 'all_appearances' format.\"\"\"\n",
        "    formatted = {}\n",
        "    for n in range(2, 7):\n",
        "        formatted[f\"{n}_grams\"] = []\n",
        "        for phrase, info in creator_results.get(n, []):\n",
        "            count = info['count']\n",
        "            if mode == 'first_appearance':\n",
        "                formatted[f\"{n}_grams\"].append({\n",
        "                    'phrase': phrase,\n",
        "                    'repetitions': count,\n",
        "                    'first_appearance': {'start': info['start'], 'end': info['end'], 'url': info['url']}\n",
        "                })\n",
        "            else:\n",
        "                formatted[f\"{n}_grams\"].append({\n",
        "                    'phrase': phrase,\n",
        "                    'repetitions': count,\n",
        "                    'appearances': info.get('appearances', [])\n",
        "                })\n",
        "    return formatted\n",
        "\n",
        "def show_results_and_save(top_results):\n",
        "    \"\"\"Formats, displays and saves the results in the correct directory structures.\"\"\"\n",
        "    if not top_results:\n",
        "        return\n",
        "\n",
        "    # Global results holders\n",
        "    global_union_first = {}\n",
        "    global_union_all = {}\n",
        "    global_interception_first = {}\n",
        "    global_interception_all = {}\n",
        "\n",
        "    for creator, results in top_results.items():\n",
        "        # Format creator results for both modes and both metrics\n",
        "        union_first = format_results(results['union'], 'first_appearance')\n",
        "        union_all = format_results(results['union'], 'all_appearances')\n",
        "        interception_first = format_results(results['intersection'], 'first_appearance')\n",
        "        interception_all = format_results(results['intersection'], 'all_appearances')\n",
        "\n",
        "        # Save global data\n",
        "        global_union_first[creator] = union_first\n",
        "        global_union_all[creator] = union_all\n",
        "        global_interception_first[creator] = interception_first\n",
        "        global_interception_all[creator] = interception_all\n",
        "\n",
        "        # Ensure creator specific directories\n",
        "        creator_first_dir = os.path.join(\"processed_files\", \"processed_n_gramas\", \"creators\", creator, \"first_appearance\")\n",
        "        creator_all_dir = os.path.join(\"processed_files\", \"processed_n_gramas\", \"creators\", creator, \"all_appearances\")\n",
        "        os.makedirs(creator_first_dir, exist_ok=True)\n",
        "        os.makedirs(creator_all_dir, exist_ok=True)\n",
        "\n",
        "        # Save creator files\n",
        "        with open(os.path.join(creator_first_dir, \"n_gramas_union.json\"), 'w', encoding='utf-8') as f:\n",
        "            json.dump(union_first, f, ensure_ascii=False, indent=2)\n",
        "        with open(os.path.join(creator_first_dir, \"n_gramas_interception.json\"), 'w', encoding='utf-8') as f:\n",
        "            json.dump(interception_first, f, ensure_ascii=False, indent=2)\n",
        "        with open(os.path.join(creator_all_dir, \"n_gramas_union.json\"), 'w', encoding='utf-8') as f:\n",
        "            json.dump(union_all, f, ensure_ascii=False, indent=2)\n",
        "        with open(os.path.join(creator_all_dir, \"n_gramas_interception.json\"), 'w', encoding='utf-8') as f:\n",
        "            json.dump(interception_all, f, ensure_ascii=False, indent=2)\n",
        "\n",
        "        print(f\"\\n{'='*50}\")\n",
        "        print(f\"Saved files for CREATOR: {creator}\")\n",
        "        print(f\"{'='*50}\")\n",
        "\n",
        "    # Save global files\n",
        "    global_first_dir = os.path.join(\"processed_files\", \"processed_n_gramas\", \"global\", \"first_appearance\")\n",
        "    global_all_dir = os.path.join(\"processed_files\", \"processed_n_gramas\", \"global\", \"all_appearances\")\n",
        "\n",
        "    with open(os.path.join(global_first_dir, \"n_gramas_union.json\"), 'w', encoding='utf-8') as f:\n",
        "        json.dump(global_union_first, f, ensure_ascii=False, indent=2)\n",
        "    with open(os.path.join(global_first_dir, \"n_gramas_interception.json\"), 'w', encoding='utf-8') as f:\n",
        "        json.dump(global_interception_first, f, ensure_ascii=False, indent=2)\n",
        "    with open(os.path.join(global_all_dir, \"n_gramas_union.json\"), 'w', encoding='utf-8') as f:\n",
        "        json.dump(global_union_all, f, ensure_ascii=False, indent=2)\n",

        "        json.dump(global_interception_all, f, ensure_ascii=False, indent=2)\n",
        "\n",
        "    print(\"\\nAll global results successfully generated and saved.\")\n",
